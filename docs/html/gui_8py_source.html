291 def poll_network():
292 current = {h for (h, _, _) in config['peers'] if h != handle}
293 newcomers = current - local_peers
294 for h in sorted(newcomers):
295 append(f"{h} joined the chat.", "#2A8940")
296 local_peers.update(newcomers)
297
298 while from_network.poll():
299 typ, src, payload = from_network.recv()
300 if typ == 'MSG':
301 append(f"{src}: {payload}", "#204EB4")
302 elif typ == 'IMG':
303 append(f"{src} sent image → {payload}", "#204EB4")
304 open_file(payload)
305 elif typ == 'LEAVE':
306 if src in already_left:
307 return
308 already_left.add(src)
309 append(f"WARNING {src} left the chat.", "#D60C0C")
310 if src in local_peers:
311 local_peers.remove(src)
312 config['peers'][:] = [p for p in config['peers'] if p[0] != src]
313
314 append(f"Welcome, {handle}!", "#000000")
